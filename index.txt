    h+=`</div>`;
  });
  h+=`</div><div class="btn-row"><button class="btn btn-orange" onclick="submitQCM()" id="submit-qcm">Submit Answers</button></div><div id="qcm-result"></div>`;
  c.innerHTML=h;
  document.querySelectorAll('.qcm-option').forEach(el=>{
    el.addEventListener('click',()=>{
      const inp=el.querySelector('input');
      document.querySelectorAll(`input[name="${inp.name}"]`).forEach(r=>r.closest('.qcm-option').classList.remove('selected'));
      inp.checked=true;el.classList.add('selected');
    });
  });
  if(ex.timer)startTimer(ex.timer,'qcm',ex);
}
function submitQCM(){
  const ex=LEVELS[state.currentLevel].exercises[state.currentExercise];
  clearTimer();let score=0;
  ex.data.forEach((q,qi)=>{
    const sel=document.querySelector(`input[name="q${qi}"]:checked`);
    const chosen=sel?parseInt(sel.value):-1;
    q.opts.forEach((_,oi)=>{
      const el=document.getElementById(`opt-${qi}-${oi}`);
      if(oi===q.a)el.classList.add('correct');
      else if(oi===chosen&&chosen!==q.a)el.classList.add('wrong');
    });
    if(chosen===q.a)score++;
  });
  document.getElementById('submit-qcm').style.display='none';
  finishExercise(score,ex.data.length,ex,'qcm-result');
}

// FINISH EXERCISE
function finishExercise(score,max,ex,resultId){
  const pass=score>=ex.passScore;
  const delta=pass?ex.successPoints:ex.penalty;
  state.totalScore+=delta;updateTopbars();
  state.levelLog.push({level:state.currentLevel+1,exercise:state.currentExercise+1,score,max,delta});
  const res=document.getElementById(resultId);
  res.innerHTML=`<div class="result-box ${pass?'result-ok':'result-fail'}">${max===10?`Code: ${score}/10`:`Score: ${score}/${max}`} — ${pass?`+${ex.successPoints} points ✅`:`${delta} points ❌`}</div>`;
  const btn=document.createElement('button');btn.className='btn btn-orange';btn.style.marginTop='12px';btn.textContent='Continue →';
  btn.onclick=()=>advanceExercise(pass);res.appendChild(btn);
}
function advanceExercise(passed){
  const lv=LEVELS[state.currentLevel];
  if(passed){unlockFragmentAndContinue();}
  else{
    const next=state.currentExercise+1;
    if(next<lv.exercises.length){state.currentExercise=next;renderExercise();}
    else{unlockFragmentAndContinue();}
  }
}
function unlockFragmentAndContinue(){
  const lvIdx=state.currentLevel;
  if(!state.fragmentsUnlocked.includes(lvIdx)){
    state.fragmentsUnlocked.push(lvIdx);
    document.getElementById('frag-num').textContent=lvIdx+1;
    document.getElementById('frag-word').textContent=FRAGMENTS[lvIdx];
    document.getElementById('frag-next-btn').onclick=()=>{
      if(lvIdx>=4){updateTopbars();showScreen('s-final');}
      else{startLevel(lvIdx+2);}
    };
    showScreen('s-fragment');
  } else {
    if(state.currentLevel>=4){updateTopbars();showScreen('s-final');}
    else{startLevel(state.currentLevel+2);}
  }
}

// TIMER

    state.timeLeft--;updateTimerDisplay();

    if(state.timeLeft<=0){clearTimer();state.totalScore-=10;updateTopbars();if(type==='qcm')submitQCM();else if(type==='match')submitMatch();}
  },1000);
}
function clearTimer(){if(state.timerInterval){clearInterval(state.timerInterval);state.timerInterval=null;}}
function updateTimerDisplay(){const el=document.getElementById('timer-display');if(el)el.textContent=fmtTime(state.timeLeft);}
